**Step 1 — Create src/components/CopyCode.astro:**

```astro
---
// CopyCode.astro — Astro island (client:visible)
// Appends a "Copiar" button to every code block in .prose on this page.
// CSS class .copy-code-btn is defined in global.css (not here) because
// scoped styles cannot reach DOM-injected elements (RESEARCH.md Pitfall 6).
// No props — queries the DOM on mount.
---

<script>
  // Runs client-side after the component becomes visible in the viewport.
  document.querySelectorAll<HTMLPreElement>('.prose pre.astro-code').forEach((pre) => {
    const btn = document.createElement('button');
    btn.className = 'copy-code-btn';
    btn.textContent = 'Copiar';
    btn.setAttribute('aria-label', 'Copiar código');

    btn.addEventListener('click', async () => {
      const code = pre.querySelector('code')?.innerText ?? pre.innerText;
      try {
        await navigator.clipboard.writeText(code);
        btn.textContent = '✓';
        btn.setAttribute('aria-label', 'Código copiado');
        setTimeout(() => {
          btn.textContent = 'Copiar';
          btn.setAttribute('aria-label', 'Copiar código');
        }, 1500);
      } catch {
        btn.textContent = 'Erro';
        setTimeout(() => {
          btn.textContent = 'Copiar';
          btn.setAttribute('aria-label', 'Copiar código');
        }, 1500);
      }
    });

    pre.appendChild(btn);
  });
</script>
```

```astro
import BaseLayout from './BaseLayout.astro';
import Header from '../components/Header.astro';

  title: string;

      <header style="margin-bottom: 32px;">

          {new Date(pubDate).toLocaleDateString('pt-BR', { dateStyle: 'long' })}
        </p>
        <h1>{title}</h1>
        {updatedDate && (
          <p style="font-family:'JetBrains Mono',monospace;font-size:11px;color:var(--texto-secundario);margin-top:8px;">
            Atualizado em {new Date(updatedDate).toLocaleDateString('pt-BR', { dateStyle: 'long' })}
          </p>
        )}
      </header>
      <slot />
    </article>
    <CopyCode client:visible />
  </div>

  <Footer slot="footer" />
</BaseLayout>
```
