    // Aggregate all unique normalized tags with post counts.
    // D-15: lowercase + dedup. Posts with empty tags silently skipped.
    // D-16: data from content.config.ts tags field.
    const posts = await getCollection('posts', ({ data }) =>

    );

    const tagCounts = new Map<string, number>();
    for (const post of posts) {
      const seen = new Set<string>();
      for (const raw of (post.data.tags ?? [])) {
        const tag = raw.toLowerCase(); // D-15: normalize
        if (!seen.has(tag)) {
          seen.add(tag);
          tagCounts.set(tag, (tagCounts.get(tag) ?? 0) + 1);
        }
      }

    // Sort by count descending, then alphabetically for equal counts.
    const sortedTags = Array.from(tagCounts.entries())
      .sort(([a, ca], [b, cb]) => cb - ca || a.localeCompare(b));

    import BaseLayout from '../../layouts/BaseLayout.astro';
    import Header from '../../components/Header.astro';

    import { getCollection } from 'astro:content';
    import type { GetStaticPaths } from 'astro';

    // D-14: dynamic route for each unique tag
    // D-15: normalize to lowercase + dedup in getStaticPaths
    // D-16: tags from content.config.ts tags field
    export const getStaticPaths: GetStaticPaths = async () => {
      const posts = await getCollection('posts', ({ data }) =>
        import.meta.env.PROD ? !data.draft : true,
      );

      // Build map: normalized tag → posts containing it
      const tagMap = new Map<string, typeof posts>();
      for (const post of posts) {
        for (const raw of (post.data.tags ?? [])) {
          const tag = raw.toLowerCase(); // D-15: normalize
          if (!tagMap.has(tag)) tagMap.set(tag, []);
          tagMap.get(tag)!.push(post);
        }
      }

      // One static path per unique tag; posts sorted newest first
      return Array.from(tagMap.entries()).map(([tag, tagPosts]) => ({
        params: { tag },
        props: {
          posts: tagPosts.sort(
            (a, b) => b.data.pubDate.valueOf() - a.data.pubDate.valueOf(),
          ),
        },
      }));
    };

    interface Props {
      posts: Awaited<ReturnType<typeof getCollection<'posts'>>>;

    const { tag } = Astro.params;
    const { posts } = Astro.props;
