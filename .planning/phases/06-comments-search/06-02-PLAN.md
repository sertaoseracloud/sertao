    ```astro
    ---

    // Search icon button + modal overlay + Pagefind default-ui initialization.
    // CSS vars for Pagefind UI theming are declared in global.css (#pagefind-search).
    // Pagefind index only exists after pnpm build — search unavailable in pnpm dev.
    ---

    <!-- Pagefind UI CSS — loaded from the CLI-generated output (version-matched) -->
    <link rel="stylesheet" href="/pagefind/pagefind-ui.css" />

    <!-- Search toggle button in header -->
    <button
      id="search-toggle"
      aria-label="Abrir busca"
      style="width:40px;height:40px;display:flex;align-items:center;justify-content:center;background:none;border:none;cursor:pointer;color:var(--nucleo-eletrico);padding:0;"
    >
      <svg
        width="20"
        height="20"
        viewBox="0 0 24 24"
        fill="none"
        stroke="currentColor"
        stroke-width="2"
        stroke-linecap="round"
        stroke-linejoin="round"
        aria-hidden="true"
      >
        <circle cx="11" cy="11" r="8"/>
        <path d="M21 21l-4.35-4.35"/>
      </svg>
    </button>
    <span class="search-kbd-hint" aria-hidden="true">Ctrl+K</span>

    <!-- Search modal overlay -->
    <div
      id="search-modal"
      role="dialog"
      aria-modal="true"
      aria-label="Busca"
      style="display:none;position:fixed;inset:0;z-index:9000;background:rgba(10,15,30,0.88);backdrop-filter:blur(4px);"
    >
      <div
        id="search-panel"
        style="position:absolute;top:15%;left:50%;transform:translateX(-50%);width:min(640px,90vw);background:var(--sub-nivel);border:1px solid var(--hairline-strong);padding:var(--space-5);"
      >
        <div id="pagefind-search"></div>
      </div>
    </div>

    <script>
      const toggleBtn = document.getElementById('search-toggle') as HTMLButtonElement;
      const modal = document.getElementById('search-modal') as HTMLElement;
      let pagefindInitialized = false;

      function openSearch(): void {
        modal.style.display = 'block';
        toggleBtn.setAttribute('aria-label', 'Fechar busca');
        if (!pagefindInitialized) {
          pagefindInitialized = true;
          // Dynamic import from CLI-generated output — not available during pnpm dev.
          import('/pagefind/pagefind-ui.js').then((mod: any) => {
            const PagefindUI = mod.PagefindUI;
            new PagefindUI({
              element: '#pagefind-search',
              showSubResults: false,
              translations: {
                placeholder: 'Buscar posts...',
                zero_results: 'Nenhum resultado para "[SEARCH_TERM]"',
              },
            });
            setTimeout(() => {
              const input = document.querySelector<HTMLInputElement>('#pagefind-search input');
              input?.focus();
            }, 50);
          });
        } else {
          const input = document.querySelector<HTMLInputElement>('#pagefind-search input');
          input?.focus();
        }
      }

      function closeSearch(): void {
        modal.style.display = 'none';
        toggleBtn.setAttribute('aria-label', 'Abrir busca');
        toggleBtn.focus();
      }

      toggleBtn.addEventListener('click', () => {
        modal.style.display === 'none' ? openSearch() : closeSearch();
      });

      document.addEventListener('keydown', (e: KeyboardEvent) => {
        if ((e.ctrlKey || e.metaKey) && e.key === 'k') {
          e.preventDefault();
          modal.style.display === 'none' ? openSearch() : closeSearch();
        }
        if (e.key === 'Escape' && modal.style.display !== 'none') {
          closeSearch();
        }
      });

      // Close modal on backdrop click (click outside the panel)
      modal.addEventListener('click', (e: MouseEvent) => {
        if (e.target === modal) closeSearch();
      });
    </script>
    ```

    Notes:
    - `import('/pagefind/pagefind-ui.js').then((mod: any)` uses `any` type to avoid TypeScript errors since the file doesn't exist at compile time (only after pnpm build).
    - The Pagefind CSS `/pagefind/pagefind-ui.css` is referenced directly from the built output, not the npm package CSS. This ensures version match with the CLI.
    - The `#pagefind-search` CSS variable overrides come from global.css (added in Plan 01, Task 1).

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User keydown → modal | Ctrl+K intercepts global keydown; e.preventDefault() stops browser default (bookmarks in Chrome) |
| Modal backdrop → close | Click events on backdrop close modal; panel click must not bubble to backdrop |
| Search input → Pagefind | User search terms sent to static index; no server-side processing |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-05 | Spoofing | Search modal aria attributes | mitigate | `role="dialog"` + `aria-modal="true"` + `aria-label="Busca"` for correct AT announcement |
| T-06-06 | Tampering | Pagefind result rendering | accept | Pagefind sanitizes HTML in results by default; no user content injected via innerHTML |
| T-06-07 | Information Disclosure | CommentsEmbed without client: directive | accept | No client: directive means no JS bundle exposure; Giscus iframe sandboxed by browser |
| T-06-08 | Elevation of Privilege | Tag chip href construction | mitigate | `tag.toLowerCase()` applied at render time ensures consistent lowercase paths matching getStaticPaths output (Pitfall 5 prevention) |
</threat_model>
